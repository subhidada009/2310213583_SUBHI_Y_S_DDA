#include <stdio.h>
#include <ctype.h>

char stack[50];
int top=-1;

void push(char x){ stack[++top]=x; }
char pop(){ return stack[top--]; }

int priority(char x){
    if(x=='+'||x=='-') return 1;
    if(x=='*'||x=='/') return 2;
    return 0;
}

int main(){
    char exp[]="A+B*C";
    int i;
    for(i=0;exp[i];i++){
        if(isalnum(exp[i]))
            printf("%c",exp[i]);
        else{
            while(top!=-1 && priority(stack[top])>=priority(exp[i]))
                printf("%c",pop());
            push(exp[i]);
        }
    }
    while(top!=-1)
        printf("%c",pop());
    return 0;
}
