#include <stdio.h>
#include <stdlib.h>

struct Node{
    int data;
    struct Node* next;
};

struct Node* last=NULL;

void insert(int x){
    struct Node* n=malloc(sizeof(struct Node));
    n->data=x;
    if(!last){
        last=n;
        n->next=n;
    } else {
        n->next=last->next;
        last->next=n;
        last=n;
    }
}

void traverse(){
    if(!last) return;
    struct Node* p=last->next;
    do{
        printf("%d ",p->data);
        p=p->next;
    }while(p!=last->next);
}

int main(){
    insert(1); insert(2); insert(3);
    traverse();
    return 0;
}
