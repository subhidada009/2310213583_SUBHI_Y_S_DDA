#include <stdio.h>
#include <stdlib.h>

struct Node{
    int data;
    struct Node *prev,*next;
};

struct Node* head=NULL;

void insert(int x){
    struct Node* n=malloc(sizeof(struct Node));
    n->data=x; n->prev=NULL; n->next=head;
    if(head) head->prev=n;
    head=n;
}

void traverse(){
    struct Node* t=head;
    while(t){ printf("%d ",t->data); t=t->next; }
}

int main(){
    insert(10); insert(20); insert(30);
    traverse();
    return 0;
}
