#include <stdio.h>

int main() {
    int mat[3][3] = {{0,0,3},{4,0,0},{0,5,0}};
    int i,j,count=0;

    for(i=0;i<3;i++)
        for(j=0;j<3;j++)
            if(mat[i][j]!=0) count++;

    int sparse[count][3];
    int k=0;
    for(i=0;i<3;i++)
        for(j=0;j<3;j++)
            if(mat[i][j]!=0){
                sparse[k][0]=i;
                sparse[k][1]=j;
                sparse[k][2]=mat[i][j];
                k++;
            }

    for(i=0;i<count;i++)
        printf("%d %d %d\n",sparse[i][0],sparse[i][1],sparse[i][2]);

    return 0;
}
